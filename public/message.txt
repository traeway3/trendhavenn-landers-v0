  <script>
    /* ⬇️ EDIT YOUR DESTINATION LINK HERE — this is the only line to change */
    const DESTINATION_URL = "https://www.trendhavenn.com/sheenus10.html";

    /* ESCAPE LOGIC — no need to edit below this line */
    const UA = navigator.userAgent || "";

    function isInAppBrowser() {
      return /BytedanceWebview|ByteLocale|musical_ly|TikTok|trill|Bytedance/i.test(UA);
    }

    function isIOS() {
      return (/iPad|iPhone|iPod/.test(UA) ||
              (navigator.platform === "MacIntel" && navigator.maxTouchPoints > 1)) &&
             !window.MSStream;
    }

    let leftApp = false;
    function markLeft() { leftApp = true; }
    document.addEventListener("visibilitychange", function () {
      if (document.hidden) markLeft();
    });
    window.addEventListener("pagehide", markLeft);

    function go(url) { window.location.href = url; }

    function openInRealBrowser() {
      if (!isInAppBrowser()) { go(DESTINATION_URL); return; }

      const bare = DESTINATION_URL.replace(/^https?:\/\//i, "");

      if (isIOS()) {
        go("x-safari-https://" + bare);
        setTimeout(function () { if (!leftApp) go(DESTINATION_URL); }, 2500);
      } else {
        go("intent://" + bare +
           "#Intent;scheme=https;action=android.intent.action.VIEW;" +
           "category=android.intent.category.BROWSABLE;package=com.android.chrome;end");
        setTimeout(function () {
          if (leftApp) return;
          go("intent://" + bare +
             "#Intent;scheme=https;action=android.intent.action.VIEW;" +
             "category=android.intent.category.BROWSABLE;end");
          setTimeout(function () { if (!leftApp) go(DESTINATION_URL); }, 1500);
        }, 1500);
      }
    }

    const btn = document.getElementById("ctaBtn");
    btn.href = DESTINATION_URL;
    btn.addEventListener("click", function (e) {
      e.preventDefault();
      openInRealBrowser();
    });
  </script>
